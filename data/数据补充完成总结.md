## 💻 使用示例

### 示例 1：加载历史战役地点

```javascript
fetch('/data/history/battles/major-battles.geojson')
  .then(res => res.json())
  .then(data => {
    map.addSource('battles', {
      type: 'geojson',
      data: data
    });
    
    // 添加战役图标
    map.addLayer({
      id: 'battle-icons',
      type: 'symbol',
      source: 'battles',
      layout: {
        'icon-image': 'battle-icon', // 需要预先加载图标
        'icon-size': 1.2,
        'text-field': ['get', 'name'],
        'text-font': ['SimHei', 'sans-serif'],
        'text-size': 14,
        'text-offset': [0, 1.5],
        'text-anchor': 'top'
      },
      paint: {
        'text-color': '#d32f2f',
        'text-halo-color': '#ffffff',
        'text-halo-width': 2
      }
    });
    
    // 添加点击事件显示详情
    map.on('click', 'battle-icons', (e) => {
      const props = e.features[0].properties;
      new maplibregl.Popup()
        .setLngLat(e.lngLat)
        .setHTML(`
          <h3>${props.name}</h3>
          <p><strong>时间</strong>: ${props.date}</p>
          <p><strong>交战双方</strong>: ${props.belligerents}</p>
          <p><strong>结果</strong>: ${props.result}</p>
          <p><strong>意义</strong>: ${props.significance}</p>
        `)
        .addTo(map);
    });
  });
```

### 示例 2：朝代疆域对比（北宋 vs 南宋）

```javascript
// 左侧地图：北宋
const map1 = new maplibregl.Map({
  container: 'map-north-song',
  style: '/data/mapstyle.json',
  center: [113, 34],
  zoom: 5,
  pitch: 60
});

// 右侧地图：南宋
const map2 = new maplibregl.Map({
  container: 'map-south-song',
  style: '/data/mapstyle.json',
  center: [118, 30],
  zoom: 5,
  pitch: 60
});

// 加载 DEM 地形
[map1, map2].forEach(map => {
  map.on('load', () => {
    map.addSource('terrain', {
      type: 'raster-dem',
      tiles: ['https://s3.amazonaws.com/elevation-tiles-prod/terrarium/{z}/{x}/{y}.png'],
      encoding: 'terrarium',
      tileSize: 256
    });
    map.setTerrain({ source: 'terrain', exaggeration: 2.5 });
    
    // 添加 hillshade
    map.addLayer({
      id: 'hillshade',
      type: 'hillshade',
      source: 'terrain',
      paint: {
        'hillshade-exaggeration': 0.5,
        'hillshade-shadow-color': '#473B24'
      }
    });
  });
});

// 加载北宋疆域
fetch('/data/history/dynasties/north-song-dynasty.geojson')
```

### 示例 3：秦统一六国态势演示（结合战役 + 疆域 + 地形）

```javascript
const map = new maplibregl.Map({
  container: 'map',
  style: '/data/mapstyle.json',
  center: [110, 35],
  zoom: 5,
  pitch: 60,
  bearing: -30
});

map.on('load', () => {
  // 1. 加载 3D 地形
  map.addSource('terrain', {
    type: 'raster-dem',
    tiles: ['https://s3.amazonaws.com/elevation-tiles-prod/terrarium/{z}/{x}/{y}.png'],
    encoding: 'terrarium'
  });
  map.setTerrain({ source: 'terrain', exaggeration: 2.5 });
  
  // 2. 加载秦朝疆域
  Promise.all([
    fetch('/data/history/dynasties/qin-dynasty.geojson').then(r => r.json()),
    fetch('/data/history/battles/major-battles.geojson').then(r => r.json()),
    fetch('/data/history/cities/ancient-capitals.geojson').then(r => r.json())
  ]).then(([qinData, battlesData, citiesData]) => {
    
    // 添加秦朝疆域
    map.addSource('qin', { type: 'geojson', data: qinData });
    map.addLayer({
      id: 'qin-territory',
      type: 'fill-extrusion',
      source: 'qin',
      paint: {
        'fill-extrusion-color': '#6a1b9a',
        'fill-extrusion-opacity': 0.6,
        'fill-extrusion-height': 5000
      }
    });
    
    // 筛选长平之战
    const changping = battlesData.features.find(f => f.properties.name === '长平之战');
    
    // 添加战役标注
    map.addSource('changping', {
      type: 'geojson',
      data: {
        type: 'FeatureCollection',
        features: [changping]
      }
    });
    
    map.addLayer({
      id: 'changping-marker',
      type: 'symbol',
      source: 'changping',
      layout: {
        'icon-image': 'battle-icon',
        'icon-size': 1.5,
        'text-field': '长平之战\n前260年',
        'text-size': 16,
        'text-anchor': 'top',
        'text-offset': [0, 1.5]
      },
      paint: {
        'text-color': '#d32f2f',
        'text-halo-color': '#ffffff',
        'text-halo-width': 2
      }
    });
    
    // 添加咸阳都城标注
    const xianyang = citiesData.features.find(f => f.properties.name === '咸阳');
    map.addSource('capital', {
      type: 'geojson',
      data: {
        type: 'FeatureCollection',
        features: [xianyang]
      }
    });
    
    map.addLayer({
      id: 'capital-marker',
      type: 'symbol',
      source: 'capital',
      layout: {
        'icon-image': 'city-icon',
        'icon-size': 1.2,
        'text-field': '咸阳\n秦都',
        'text-size': 14,
        'text-anchor': 'top',
        'text-offset': [0, 1.5]
      },
      paint: {
        'text-color': '#1976d2',
        'text-halo-color': '#ffffff',
        'text-halo-width': 2
      }
    });
  });
});
```

## 🔄 数据溯源与质量

### 数据来源
1. **朝代疆域**（CHGIS 真实数据）：
   - 秦/西汉/东汉/唐/元/明/清：来自 CHGIS V6 数据库
   - 北宋/南宋：基于谭其骧《中国历史地图集》手工绘制

   - 坐标来源：CHGIS 地名数据库交叉验证
   - 属性信息：《史记》《资治通鉴》及中学历史教材

3. **历史名城**：
   - 坐标来源：CHGIS + 现代城市中心点
   - 属性信息：历史地图集 + 地方志

4. **战略要地**：
   - 坐标来源：历史文献 + 实地测绘数据
   - 关隘位置：结合 DEM 地形验证
