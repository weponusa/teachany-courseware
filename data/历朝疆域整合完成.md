## 💻 使用示例

### 示例 1: 显示秦朝疆域

```javascript
fetch('/data/history/dynasties/qin-dynasty.geojson')
  .then(res => res.json())
  .then(data => {
    console.log('秦朝行政区数:', data.features.length); // 183
    
    map.addSource('qin', {
      type: 'geojson',
      data: data
    });
    
    map.addLayer({
      id: 'qin-fill',
      type: 'fill',
      source: 'qin',
      paint: {
        'fill-color': '#6a1b9a',
        'fill-opacity': 0.3
      }
    });
    
    map.addLayer({
      id: 'qin-border',
      type: 'line',
      source: 'qin',
      paint: {
        'line-color': '#6a1b9a',
        'line-width': 2
      }
    });
  });
```

### 示例 2: 朝代疆域对比（秦 vs 清）

```javascript
const dynasties = [
  { id: 'qin', file: 'qin-dynasty.geojson', color: '#6a1b9a', name: '秦朝' },
  { id: 'qing', file: 'qing-dynasty.geojson', color: '#d32f2f', name: '清朝' }
];

// 加载两个朝代数据
Promise.all(
  dynasties.map(d => 
    fetch(`/data/history/dynasties/${d.file}`).then(r => r.json())
  )
).then(dataArray => {
  dynasties.forEach((dynasty, index) => {
    const data = dataArray[index];
    
    map.addSource(dynasty.id, {
      type: 'geojson',
      data: data
    });
    
    map.addLayer({
      id: `${dynasty.id}-fill`,
      type: 'fill',
      source: dynasty.id,
      paint: {
        'fill-color': dynasty.color,
        'fill-opacity': 0.2
      }
    });
    
    map.addLayer({
      id: `${dynasty.id}-border`,
      type: 'line',
      source: dynasty.id,
      paint: {
        'line-color': dynasty.color,
        'line-width': 2
      }
    });
  });
  
  // 显示图例
  console.log('已加载朝代:', dynasties.map(d => d.name).join(' vs '));
});
```

### 示例 3: 统计朝代行政区数量变化

```javascript
const dynastyFiles = [
  'qin-dynasty.geojson',      // 秦
  'west-han-dynasty.geojson', // 西汉

  'yuan-dynasty.geojson',     // 元
  'ming-dynasty.geojson',     // 明
  'qing-dynasty.geojson'      // 清
];

Promise.all(
  dynastyFiles.map(file =>
    fetch(`/data/history/dynasties/${file}`).then(r => r.json())

  const stats = dataArray.map((data, i) => ({
    dynasty: dynastyFiles[i].replace('-dynasty.geojson', ''),
    count: data.features.length
  }));
  
  console.table(stats);
  // 输出:
  // qin          183
  // west-han     233
  // east-han     440
  // tang         225
  // yuan         237
  // ming         233
  // qing         627
  
  // 绘制柱状图或折线图...
});
```
